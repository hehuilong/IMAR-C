00086 //----------------------------------------------------------------------
00087 //  computeDistortion
00088 //      This procedure computes the total and individual distortions for
00089 //      a set of center points.  It invokes getNeighbors() on the
00090 //      kc-tree for the point set,which computes the values of weights,
00091 //      sums, and sumSqs, from which the distortion is computed as
00092 //      follows.
00093 //
00094 //      Distortion Computation:
00095 //      -----------------------
00096 //      Assume that some center has been fixed (indexed by j in the code
00097 //      below).  Let SUM_i denote a summation over all (wgt[j])
00098 //      neighbors of the given center.  The data points (p[i]) and
00099 //      center points (c[j]) are vectors, and the product of two vectors
00100 //      means the dot product (u*v = (u.v), u^2 = (u.u)).  The
00101 //      distortion for a single center j, denoted dists[j], is defined
00102 //      to be the sum of squared distances from each point to its
00103 //      closest center,  That is:
00104 //
00105 //          dists[j] = SUM_i (p[i] - c[j])^2
00106 //              = SUM_i (p[i]^2 - 2*c[j]*p[i] + c[j]^2)
00107 //              = SUM_i p[i]^2 - 2*c[j]*SUM_i p[i] + wgt[j]*c[j]^2
00108 //              = sumSqs[j] - 2*(c[j].sums[j]) + wgt[j]*(c[j]^2)
00109 //
00110 //      Thus the individual distortion can be computed from these
00111 //      quantities.  The total distortion is the sum of the individual
00112 //      distortions.
00113 //----------------------------------------------------------------------
00114 
00115 void KMfilterCenters::computeDistortion() // compute distortions
00116 {
00117     // *kmOut << "------------------------------Computing Distortions" << endl;
00118     KCtree* t = getData().getKcTree();
00119     assert(t != NULL);                          // tree better exist
00120     t->getNeighbors(*this);                     // get neighbors
00121     double totDist = 0;
00122     for (int j = 0; j < kCtrs; j++) {
00123         double cDotC = 0;                       // init: (c[j] . c[j])
00124         double cDotS = 0;                       // init: (c[j] . sum[j])
00125         for (int d = 0; d < getDim(); d++) {    // compute dot products
00126             cDotC += ctrs[j][d] * ctrs[j][d];
00127             cDotS += ctrs[j][d] * sums[j][d];
00128         }
00129                                                 // final distortion
00130         dists[j] = sumSqs[j] - 2*cDotS + weights[j]*cDotC;
00131         totDist += dists[j];
00132     }
00133     currDist = totDist;                         // save total distortion
00134 
00135     validate();                                 // distortions are now valid
00136 }

00158 //----------------------------------------------------------------------
00159 //  moveToCentroid
00160 //      This procedure moves each center point to the centroid of its
00161 //      associated cluster.  We call computeDistortion() if necessary to
00162 //      compute the weights and sums.  The centroid is the weighted
00163 //      average of the sum of neighbors.  Thus the 
00164 //
00165 //          ctrs[j] = sums[j] / weights[j].
00166 //
00167 //      We generally allow a dampening factor on the motion, which is a
00168 //      floating quantity between 0 (full dampening) and 1 (no
00169 //      dampening).  Given the dampening factor df, the above formula
00170 //      is:
00171 //          ctrs[j] = (1-df) * ctrs[j] + df * sums[j]/ weights[j]
00172 //
00173 //----------------------------------------------------------------------
00174 
00175 void KMfilterCenters::moveToCentroid()  // move center to cluster centroid
00176 {
00177     if (!valid) computeDistortion();            // compute sums if needed
00178     for (int j = 0; j < kCtrs; j++) {
00179         int wgt = weights[j];                   // weight of this center
00180         if (wgt > 0) {                          // update only if weight > 0
00181             for (int d = 0; d < getDim(); d++) {
00182                 ctrs[j][d] = (1 - dampFactor) * ctrs[j][d] +
00183                                 dampFactor * sums[j][d]/wgt;
00184             }
00185         }
00186     }
00187     invalidate();                               // distortions now invalid
00188 }

00191 //  swapOneCenter
00192 //      Swaps one center point with a sample point.  Optionally we make
00193 //      sure that the new point is not a duplicate of any of the centers
00194 //      (including the point being replaced).
00195 //----------------------------------------------------------------------
00196 void KMfilterCenters::swapOneCenter(            // swap one center
00197     bool allowDuplicate)                        // allow duplicate centers
00198 {
00199     int rj = kmRanInt(kCtrs);                   // index of center to replace
00200     int dim = getDim();
00201     KMpoint p = kmAllocPt(dim);                 // alloc replacement point
00202     pts->sampleCtr(p);                          // sample a replacement
00203     if (!allowDuplicate) {                      // duplicates not allowed?
00204         bool dupFound;                          // was a duplicate found?
00205         do {                                    // repeat until successful
00206             dupFound = false;
00207             for (int j = 0; j < kCtrs; j++) {   // search for duplicates
00208                 if (kmEqualPts(dim, p, ctrs[j])) {
00209                     dupFound = true;
00210                     pts->sampleCtr(p);          // try again
00211                     break;
00212                 }
00213             }
00214         } while (dupFound);
00215     }
00216     kmCopyPt(dim, p, ctrs[rj]);                 // copy sampled point
00217     if (kmStatLev >= STEP) {                    // output swap info
00218         *kmOut << "\tswapping: ";
00219         kmPrintPt(p, getDim(), true);
00220         *kmOut << "<-->Center[" << rj << "]\n";
00221     }
00222     kmDeallocPt(p);                             // deallocate point storage
00223     invalidate();                               // distortions now invalid
00224 }
