00088 //----------------------------------------------------------------------
00089 //  Main program
00090 //----------------------------------------------------------------------
00091 int main(int argc, char **argv)
00092 {
00093     getArgs(argc, argv);                        // read command-line arguments
00094     term.setAbsMaxTotStage(stages);             // set number of stages
00095 
00096     KMdata dataPts(dim, maxPts);                // allocate data storage
00097     int nPts = 0;                               // actual number of points
00098                                                 // generate points
00099     if (dataIn != NULL) {                       // read points from file
00100         while (nPts < maxPts && readPt(*dataIn, dataPts[nPts])) nPts++;
00101     }
00102     else {                                      // generate points randomly
00103         nPts = maxPts;
00104         kmClusGaussPts(dataPts.getPts(), nPts, dim, k);
00105     }
00106 
00107     cout << "Data Points:\n";                   // echo data points
00108     for (int i = 0; i < nPts; i++)
00109         printPt(cout, dataPts[i]);
00110 
00111     dataPts.setNPts(nPts);                      // set actual number of pts
00112     dataPts.buildKcTree();                      // build filtering structure
00113 
00114     KMfilterCenters ctrs(k, dataPts);           // allocate centers
00115 
00116                                                 // run each of the algorithms
00117     cout << "\nExecuting Clustering Algorithm: Lloyd's\n";
00118     KMlocalLloyds kmLloyds(ctrs, term);         // repeated Lloyd's
00119     ctrs = kmLloyds.execute();                  // execute
00120     printSummary(kmLloyds, dataPts, ctrs);      // print summary
00121 
00122     cout << "\nExecuting Clustering Algorithm: Swap\n";
00123     KMlocalSwap kmSwap(ctrs, term);             // Swap heuristic
00124     ctrs = kmSwap.execute();
00125     printSummary(kmSwap, dataPts, ctrs);
00126 
00127     cout << "\nExecuting Clustering Algorithm: EZ-Hybrid\n";
00128     KMlocalEZ_Hybrid kmEZ_Hybrid(ctrs, term);   // EZ-Hybrid heuristic
00129     ctrs = kmEZ_Hybrid.execute();
00130     printSummary(kmEZ_Hybrid, dataPts, ctrs);
00131 
00132     cout << "\nExecuting Clustering Algorithm: Hybrid\n";
00133     KMlocalHybrid kmHybrid(ctrs, term);         // Hybrid heuristic
00134     ctrs = kmHybrid.execute();
00135     printSummary(kmHybrid, dataPts, ctrs);
00136
