00291             float min_var, float max_var, float max_dis)

00293         int size = track.size();
00294         for(int i = 0; i < size; i++) {

00299         mean_y /= size;
00300

00302                 track[i].x -= mean_x;

00305                 var_y += track[i].y*track[i].y;

00308         var_y /= size;
00309         var_x = sqrt(var_x);
00310         var_y = sqrt(var_y);
00311         // remove static trajectory
00312         if(var_x < min_var && var_y < min_var)
00313                 return 0;
00314         // remove random trajectory
00315         if( var_x > max_var || var_y > max_var )
00316                 return 0;
00317 
00318         for(int i = 1; i < size; i++) {
00319                 float temp_x = track[i].x - track[i-1].x;
00320                 float temp_y = track[i].y - track[i-1].y;
00321                 length += sqrt(temp_x*temp_x+temp_y*temp_y);
00322                 track[i-1].x = temp_x;
00323                 track[i-1].y = temp_y;
00324         }
00325 
00326         float len_thre = length*0.7;
00327         for( int i = 0; i < size-1; i++ ) {
00328                 float temp_x = track[i].x;
00329                 float temp_y = track[i].y;
00330                 float temp_dis = sqrt(temp_x*temp_x + temp_y*temp_y);
00331                 if( temp_dis > max_dis && temp_dis > len_thre )
00332                         return 0;
00333         }
00334

00336         // normalize the trajectory
00337         for(int i = 0; i < size-1; i++) {
