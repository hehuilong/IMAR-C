00035 KMdist kmDist(                  // interpoint squared distance
00036     int                 dim,
00037     KMpoint             p,
00038     KMpoint             q)
00039 {
00040     register int d;
00041     register KMcoord diff;
00042     register KMcoord dist;
00043 
00044     dist = 0;
00045     for (d = 0; d < dim; d++) {
00046         diff = p[d] - q[d];
00047         dist = KM_SUM(dist, KM_POW(diff));
00048     }
00049     return dist;
00050 }

00147 //----------------------------------------------------------------------
00148 //  Methods for orthogonal rectangles:
00149 //      kmAssignRect() assigns the coordinates of one rectangle to
00150 //      another.  The two rectangles must have the same dimension
00151 //      (and it is not possible to test this here).
00152 //
00153 //      inside() returns true if a point lies inside the (closed)
00154 //      rectangle and false otherwise.
00155 //      
00156 //      expand(d,x,r) expands this rectangle by a factor of x, centrally
00157 //      about its origin and stores the resulting rectangle in r.
00158 //----------------------------------------------------------------------
00159    
00160                                                 // assign one rect to another
00161 void kmAssignRect(int dim, KMorthRect &dest, const KMorthRect &source)
00162 {
00163     for (int i = 0; i < dim; i++) {
00164         dest.lo[i] = source.lo[i];
00165         dest.hi[i] = source.hi[i];
00166     }
00167 }
00168                                                 // is point inside rectangle?
00169 bool KMorthRect::inside(int dim, KMpoint p)
00170 {
00171     for (int i = 0; i < dim; i++) {
00172         if (p[i] < lo[i] || p[i] > hi[i]) return false;
00173     }
00174     return true;
00175 }
00176                                                 // expand by factor x
00177 void KMorthRect::expand(int dim, double x, KMorthRect r)
00178 {
00179     for (int i = 0; i < dim; i++) {
00180         KMcoord wid = hi[i] - lo[i];
00181         r.lo[i] = lo[i] - (wid/2)*(x - 1);
00182         r.hi[i] = hi[i] + (wid/2)*(x - 1);
00183     }
00184 }
