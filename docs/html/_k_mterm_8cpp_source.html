00059 int KMterm::maxStage(const double param[KM_TERM_VEC_LEN],
00060                                 int k, int n) const
00061 {
00062     double count = param[KM_TERM_CONST];
00063     if (param[KM_TERM_POW] != 0) {
00064         double sum = param[KM_TERM_LIN_K] * k + param[KM_TERM_LIN_N] * n;
00065         count += pow(sum, param[KM_TERM_POW]);
00066     }
00067     assert(count >= 0 && count <= INT_MAX);     // should be positive integer
00068     if (count <= 0) count = INT_MAX;            // 0 means infinity
00069     return int(count);
00070 }
