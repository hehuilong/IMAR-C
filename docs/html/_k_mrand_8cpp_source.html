00033 //------------------------------------------------------------------------
00034 //      kmRan0 - (safer) uniform random number generator
00035 //
00036 //      The code given here is taken from "Numerical Recipes in C" by
00037 //      William Press, Brian Flannery, Saul Teukolsky, and William
00038 //      Vetterling. The task of the code is to do an additional randomizing
00039 //      shuffle on the system-supplied random number generator to make it
00040 //      safer to use. 
00041 //
00042 //      Returns a uniform deviate between 0.0 and 1.0 using the
00043 //      system-supplied routine "random()". Set kmIdum to any negative value
00044 //      to initialise or reinitialise the sequence.
00045 //------------------------------------------------------------------------
00046 
00047 static double kmRan0()
00048 {
00049     int j;
00050 
00051     static double y, maxran, v[98];     // The exact number 98 is unimportant
00052     static int iff = 0;
00053 
00054     // As a precaution against misuse, we will always initialize on the first
00055     // call, even if "kmIdum" is not set negative. Determine "maxran", the
00056     // next integer after the largest representable value of type int. We
00057     // assume this is a factor of 2 smaller than the corresponding value of
00058     // type unsigned int. 
00059 
00060     if (kmIdum < 0 || iff == 0) {       // initialize
00061                 /* compute maximum random number */
00062 #ifdef WIN32                            // Microsoft Visual C++
00063         maxran = RAND_MAX;
00064 #else
00065         unsigned i, k;
00066         i = 2;
00067         do {
00068             k = i;
00069             i <<= 1;
00070         } while (i);
00071         maxran = (double) k;
00072 #endif
00073         iff = 1;
00074   
00075         srandom(kmIdum);
00076         kmIdum = 1;
00077 
00078         for (j = 1; j <= 97; j++)       // exercise the system routine
00079             random();                   // (value intentionally ignored)
00080 
00081         for (j = 1; j <= 97; j++)       // Then save 97 values and a 98th
00082             v[j] = random();
00083         y = random();
00084      }
00085 
00086     // This is where we start if not initializing. Use the previously saved
00087     // random number y to get an index j between 1 and 97. Then use the
00088     // corresponding v[j] for both the next j and as the output number. */
00089 
00090     j = 1 + (int) (97.0 * (y / maxran));
00091     y = v[j];
00092     v[j] = random();                    // Finally, refill the table entry
00093                                         // with the next random number from
00094                                         // "random()" 
00095     return(y / maxran);
00096 }

00105     int                 n)
00106 {
00107     int r = (int) (kmRan0()*n);

00123 //------------------------------------------------------------------------
00124 //  kmRanGauss - Gaussian random number generator
00125 //      Returns a normally distributed deviate with zero mean and unit
00126 //      variance, using kmRan0() as the source of uniform deviates.
00127 //------------------------------------------------------------------------
00128 
00129 static double kmRanGauss()
00130 {
00131     static int iset=0;
00132     static double gset;
00133 
00134     if (iset == 0) {                    // we don't have a deviate handy
00135         double v1, v2;
00136         double r = 2.0;
00137         while (r >= 1.0) {
00138             //------------------------------------------------------------
00139             // Pick two uniform numbers in the square extending from -1 to
00140             // +1 in each direction, see if they are in the circle of radius
00141             // 1.  If not, try again 
00142             //------------------------------------------------------------
00143             v1 = kmRanUnif(-1, 1);
00144             v2 = kmRanUnif(-1, 1);
00145             r = v1 * v1 + v2 * v2;
00146         }
00147         double fac = sqrt(-2.0 * log(r) / r);
00148         //-----------------------------------------------------------------
00149         // Now make the Box-Muller transformation to get two normal
00150         // deviates.  Return one and save the other for next time.
00151         //-----------------------------------------------------------------
00152         gset = v1 * fac;
00153         iset = 1;                       // set flag
00154         return v2 * fac;
00155     }
00156     else {                              // we have an extra deviate handy
00157         iset = 0;                       // so unset the flag
00158         return gset;                    // and return it
00159     }
00160 }

00162 //------------------------------------------------------------------------
00163 //  kmRanLaplace - Laplacian random number generator
00164 //      Returns a Laplacian distributed deviate with zero mean and
00165 //      unit variance, using kmRan0() as the source of uniform deviates. 
00166 //
00167 //              prob(x) = b/2 * exp(-b * |x|).
00168 //
00169 //      b is chosen to be sqrt(2.0) so that the variance of the Laplacian
00170 //      distribution [2/(b^2)] becomes 1. 
00171 //------------------------------------------------------------------------
00172 
00173 static double kmRanLaplace() 
00174 {
00175     const double b = 1.4142136;
00176 
00177     double laprand = -log(kmRan0()) / b;
00178     double sign = kmRan0();
00179     if (sign < 0.5) laprand = -laprand;
00180     return(laprand);
00181 }

00237 //  kmCoGaussPts - Generate correlated Gaussian distributed points
00238 //      Generates a Gauss-Markov distribution of zero mean and unit
00239 //      variance.
00240 //----------------------------------------------------------------------
00241 
00242 void kmCoGaussPts(              // correlated-Gaussian distribution
00243         KMpointArray    pa,             // point array (modified)
00244         int             n,              // number of points
00245         int             dim,            // dimension

00249     for (int i = 0; i < n; i++) {

00255         }

00260 //  kmCoLaplacePts - Generate correlated Laplacian distributed points
00261 //      Generates a Laplacian-Markov distribution of zero mean and unit
00262 //      variance.
00263 //----------------------------------------------------------------------
00264 
00265 void kmCoLaplacePts(            // correlated-Laplacian distribution

00267         int             n,              // number of points
00268         int             dim,            // dimension
00269         double          correlation)    // correlation

00273

00289 //----------------------------------------------------------------------
00290 //  kmClusGaussPts - Generate clusters of Gaussian distributed points
00291 //      Cluster centers are uniformly distributed over [-1,1], and the
00292 //      standard deviation within each cluster is fixed.
00293 //
00294 //      Note: Once cluster centers have been set, they are not changed,
00295 //      unless new_clust = true.  This is so that subsequent calls generate
00296 //      points from the same distribution.  It follows, of course, that any
00297 //      attempt to change the dimension or number of clusters without
00298 //      generating new clusters is asking for trouble.
00299 //
00300 //      Note: Cluster centers are not generated by a call to uniformPts().
00301 //      Although this could be done, it has been omitted for
00302 //      compatibility with kmClusGaussPts() in the colored version,
00303 //      rand_c.cc.
00304 //
00305 //      As a side effect we compute the cluster separation. This is
00306 //      defined to be min_{i != j} dist(cc[i],cc[j])/std, where cc[i] is
00307 //      the i-th cluster center and std is the global deviation.  It is
00308 //      defined to be the square root of the trace of the covariance
00309 //      matrix, which is equivalent to the std_dev for each coordinate
00310 //      times sqrt(d).
00311 //----------------------------------------------------------------------
00312 static KMpointArray cgClusters = NULL;  // cluster storage
00313 
00314 void kmClusGaussPts(            // clustered-Gaussian distribution
00315         KMpointArray    pa,             // point array (modified)
00316         int             n,              // number of points
00317         int             dim,            // dimension
00318         int             n_col,          // number of colors
00319         bool            new_clust,      // generate new clusters.
00320         double          std_dev,        // standard deviation within clusters
00321         double*         clus_sep)       // cluster separation (returned)
00322 {
00323     if (cgClusters == NULL || new_clust) {// need new cluster centers
00324         if (cgClusters != NULL)         // clusters already exist
00325             kmDeallocPts(cgClusters);   // get rid of them
00326         cgClusters = kmAllocPts(n_col, dim);
00327                                         // generate cluster center coords
00328         for (int i = 0; i < n_col; i++) {
00329             for (int d = 0; d < dim; d++) {
00330                 cgClusters[i][d] = (KMcoord) kmRanUnif(-1,1);
00331             }
00332         }
00333     }
00334 
00335     double minDist = double(dim);       // minimum inter-center sq'd distance
00336     for (int i = 0; i < n_col; i++) {   // compute minimum separation
00337         for (int j = i+1; j < n_col; j++) {
00338             double dist = kmDist(dim, cgClusters[i], cgClusters[j]);
00339             if (dist < minDist) minDist = dist;
00340         }
00341     }
00342                                         // cluster separation
00343     if (clus_sep != NULL)
00344         *clus_sep = sqrt(minDist)/(sqrt(double(dim))*std_dev);
00345 
00346     for (int i = 0; i < n; i++) {
00347         int c = kmRanInt(n_col);        // generate cluster index
00348         for (int d = 0; d < dim; d++) {
00349           pa[i][d] = (KMcoord) (std_dev*kmRanGauss() + cgClusters[c][d]);
00350         }
00351     }
00352 }

00360 //  kmClusOrthFlats - points clustered along orthogonal flats
00361 //
00362 //      This distribution consists of a collection points clustered
00363 //      among a collection of low dimensional flats (hyperplanes) in
00364 //      the hypercube [-1,1]^d.  A set of n_col orthogonal flats are
00365 //      generated, each whose dimension is a random number between 1
00366 //      and max_dim.  The points are evenly distributed among the clusters.
00367 //      For each cluster, we generate points uniformly distributed along
00368 //      the flat within the hypercube.
00369 //
00370 //      This is done as follows.  Each cluster is defined by a d-element
00371 //      control vector whose components are either:
00372 //      
00373 //              CO_FLAG indicating that this component is to be generated
00374 //                      uniformly in [-1,1],
00375 //              x       a value other than CO_FLAG in the range [-1,1],
00376 //                      which indicating that this coordinate is to be
00377 //                      generated as x plus a Gaussian random deviation
00378 //                      with the given standard deviation.
00379 //                      
00380 //      The number of zero components is the dimension of the flat, which
00381 //      is a random integer in the range from 1 to max_dim.  The points
00382 //      are disributed between clusters in nearly equal sized groups.
00383 //
00384 //      Note: Once cluster centers have been set, they are not changed,
00385 //      unless new_clust = true.  This is so that subsequent calls generate
00386 //      points from the same distribution.  It follows, of course, that any
00387 //      attempt to change the dimension or number of clusters without
00388 //      generating new clusters is asking for trouble.
00389 //
00390 //      To make this a bad scenario at query time, query points should be
00391 //      selected from a different distribution, e.g. uniform or Gaussian.
00392 //
00393 //      We use a little programming trick to generate groups of roughly
00394 //      equal size.  If n is the total number of points, and n_col is
00395 //      the number of clusters, then the c-th cluster (0 <= c < n_col)
00396 //      is given floor((n+c)/n_col) points.  It can be shown that this
00397 //      will exactly consume all n points.
00398 //
00399 //----------------------------------------------------------------------
00400 
00401 void kmClusOrthFlats(           // clustered along orthogonal flats
00402         KMpointArray    pa,             // point array (modified)
00403         int             n,              // number of points
00404         int             dim,            // dimension
00405         int             n_col,          // number of colors
00406         bool            new_clust,      // generate new clusters.
00407         double          std_dev,        // standard deviation within clusters
00408         int             max_dim)        // maximum dimension of the flats
00409 {
00410     const double CO_FLAG = 999;                 // special flag value
00411     static KMpointArray control = NULL; // control vectors
00412 
00413     if (control == NULL || new_clust) {         // need new cluster centers
00414         if (control != NULL) {                  // clusters already exist
00415             kmDeallocPts(control);              // get rid of them
00416         }
00417         control = kmAllocPts(n_col, dim);
00418 
00419         for (int c = 0; c < n_col; c++) {       // generate clusters
00420             int n_dim = 1 + kmRanInt(max_dim);  // number of dimensions in flat
00421             for (int d = 0; d < dim; d++) {     // generate side locations
00422                                                 // prob. of picking next dim
00423                 double Prob = ((double) n_dim)/((double) (dim-d));
00424                 if (kmRan0() < Prob) {          // add this one to flat
00425                     control[c][d] = CO_FLAG;    // flag this entry
00426                     n_dim--;                    // one fewer dim to fill
00427                 }
00428                 else {                          // don't take this one
00429                     control[c][d] = kmRanUnif(-1,1);// random value in [-1,1]
00430                 }
00431             }
00432         }
00433     }
00434 
00435     int next = 0;                               // next slot to fill
00436     for (int c = 0; c < n_col; c++) {           // generate clusters
00437         int pick = (n+c)/n_col;                 // number of points to pick
00438         for (int i = 0; i < pick; i++) {
00439             for (int d = 0; d < dim; d++) {
00440                 if (control[c][d] == CO_FLAG)   // dimension on flat
00441                     pa[next][d] = (KMcoord) kmRanUnif(-1,1);
00442                 else                            // dimension off flat
00443                     pa[next][d] =
00444                         (KMcoord) (std_dev*kmRanGauss() + control[c][d]);
00445             }
00446             next++;
00447         }
00448     }
00449 }

00452 //  kmClusEllipsoids - points clustered around ellipsoids
00453 //
00454 //      This distribution consists of a collection points clustered
00455 //      among a collection of low dimensional ellipsoids in the
00456 //      hypercube [-1,1]^d.  The objective is to model distributions
00457 //      in which the points are distributed in lower dimensional
00458 //      subspaces, and within this lower dimensional space the points
00459 //      are distributed with a Gaussian distribution (with no
00460 //      correlation between the dimensions).
00461 //
00462 //      The distribution is given the number of clusters or "colors"
00463 //      (n_col), maximum number of dimensions (max_dim) of the lower
00464 //      dimensional subspace, a "small" standard deviation (std_dev_small),
00465 //      and a "large" standard deviation range (std_dev_lo, std_dev_hi).
00466 //
00467 //      The algorithm generates n_col cluster centers uniformly from the
00468 //      hypercube [-1,1]^d.  For each cluster, it selects the dimension
00469 //      of the subspace as a random number r between 1 and max_dim.
00470 //      These are the dimensions of the ellipsoid.  Then it generates
00471 //      a d-element std dev vector whose entries are the standard
00472 //      deviation for the coordinates of each cluster in the distribution.
00473 //      Among the d-element control vector, r randomly chosen values are
00474 //      chosen uniformly from the range [std_dev_lo, std_dev_hi].  The
00475 //      remaining values are set to std_dev_small.
00476 //
00477 //      Note that kmClusGaussPts is a special case of this in which
00478 //      max_dim = 0, and std_dev = std_dev_small.
00479 //
00480 //      If the flag new_clust is set, then new cluster centers are
00481 //      generated.
00482 //
00483 //----------------------------------------------------------------------
00484 
00485 void kmClusEllipsoids(          // clustered around ellipsoids
00486         KMpointArray    pa,             // point array (modified)
00487         int             n,              // number of points
00488         int             dim,            // dimension
00489         int             n_col,          // number of colors
00490         bool            new_clust,      // generate new clusters.
00491         double          std_dev_small,  // small standard deviation
00492         double          std_dev_lo,     // low standard deviation for ellipses
00493         double          std_dev_hi,     // high standard deviation for ellipses
00494         int             max_dim)        // maximum dimension of the flats
00495 {
00496     static KMpointArray clusters = NULL;        // cluster centers
00497     static KMpointArray stdDev = NULL;          // standard deviations
00498 
00499     if (clusters == NULL || new_clust) {        // need new cluster centers
00500         if (clusters != NULL)                   // clusters already exist
00501             kmDeallocPts(clusters);             // get rid of them
00502         if (stdDev != NULL)                     // std deviations already exist
00503             kmDeallocPts(stdDev);               // get rid of them
00504 
00505         clusters = kmAllocPts(n_col, dim);      // alloc new clusters and devs
00506         stdDev   = kmAllocPts(n_col, dim);
00507 
00508         for (int i = 0; i < n_col; i++) {       // gen cluster center coords
00509             for (int d = 0; d < dim; d++) {
00510                 clusters[i][d] = (KMcoord) kmRanUnif(-1,1);
00511             }
00512         }
00513         for (int c = 0; c < n_col; c++) {       // generate cluster std dev
00514             int n_dim = 1 + kmRanInt(max_dim);  // number of dimensions in flat
00515             for (int d = 0; d < dim; d++) {     // generate std dev's
00516                                                 // prob. of picking next dim
00517                 double Prob = ((double) n_dim)/((double) (dim-d));
00518                 if (kmRan0() < Prob) {          // add this one to ellipse
00519                                                 // generate random std dev
00520                     stdDev[c][d] = kmRanUnif(std_dev_lo, std_dev_hi);
00521                     n_dim--;                    // one fewer dim to fill
00522                 }
00523                 else {                          // don't take this one
00524                     stdDev[c][d] = std_dev_small;// use small std dev
00525                 }
00526             }
00527         }
00528     }
00529 
00530     int next = 0;                               // next slot to fill
00531     for (int c = 0; c < n_col; c++) {           // generate clusters
00532         int pick = (n+c)/n_col;                 // number of points to pick
00533         for (int i = 0; i < pick; i++) {
00534             for (int d = 0; d < dim; d++) {
00535                 pa[next][d] = (KMcoord)
00536                         (stdDev[c][d]*kmRanGauss() + clusters[c][d]);
00537             }
00538             next++;
00539         }
00540     }
00541 }

00543 //----------------------------------------------------------------------
00544 //  kmMultiClus - multi-sized clusters
00545 //      This distribution is designed to be a challenge for clustering
00546 //      algorithm.  It consists of a clusters of varying sizes, located
00547 //      within the cube [-1,1]^d.  The cluster centers are uniformly
00548 //      distributed.  We generate clusters one by one.  For each
00549 //      cluster, the size of the cluster m is chosen so that the
00550 //      probability of generating a cluster of size 2^i is 1/2^i.
00551 //
00552 //      We want each cluster to have a similar total squared distortion
00553 //      (variance).  Thus a group of size m is generated from a gaussian
00554 //      distribution with a standard deviation of B*sqrt(1/m).  We call
00555 //      B the base standard deviation, which is given as an argument.
00556 //
00557 //      The expected number of clusters is (lg n)-2, but there is a very
00558 //      high variance here.  The total distortion in each dimension then
00559 //      is B^2*((lg n)-2) (but don't trust me here, since this is just
00560 //      based on back-of- the-envelope computations).  Hence the expected
00561 //      average distortion, for n points in dim d would be:
00562 //
00563 //              d*B^2*((log n)-2)/n
00564 //
00565 //      The reason that this is challenging is that each cluster has an
00566 //      equal claim to a center (since distortions are similar) but many
00567 //      sampling based methods will favor placing clusters in the
00568 //      relatively few clusters with a large number of points.
00569 //----------------------------------------------------------------------
00570 
00571 void kmMultiClus(               // multi-sized clusters
00572         KMpointArray    pa,             // point array (modified)
00573         int             n,              // number of points
00574         int             dim,            // dimension
00575         int             &k,             // number of clusters (returned)
00576         double          base_dev)       // base standard deviation
00577 {
00578     int next = 0;                       // next point in array
00579     int nSamp = 0;                      // number of points sampled
00580     k = 0;                              // number of clusters generated
00581     KMpoint clusCenter = kmAllocPt(dim); // allocate center storage
00582     while (nSamp < n) {                 // until we have sampled enough
00583         int remain = n - nSamp;         // number remaining to sample
00584         int clusSize = 2;
00585                                         // repeatedly double cluster size
00586                                         // with prob 1/2
00587         while ((clusSize < remain) && (kmRan0() < 0.5))
00588             clusSize *= 2;
00589                                         // don't exceed upper limit
00590         if (clusSize > remain) clusSize = remain;
00591 
00592                                         // generate center uniformly
00593         for (int d = 0; d < dim; d++) {
